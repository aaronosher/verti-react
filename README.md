## Install

```bash
npm install --save verti-react
```

```bash
npm add verti-react
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'verti-react'
import 'verti-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
